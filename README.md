## Install

```bash
npm install --save stacked-carousel
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'stacked-carousel'
import 'stacked-carousel/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
